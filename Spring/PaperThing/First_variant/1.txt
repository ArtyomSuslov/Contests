    class A
    {
    public: 
        [virtual] ~A() {std::cout << "~A()" << std::endl;}
    private:
        int a;
    };

    class B : A 
    {
    public:
        ~B() {std::cout << "~B()" << std::endl;}
    private:
    }

    A *obj = new B();
    delete obj; 

    Если деструктор виртуальный, то вывод будет ~B() ~A()
    Если деструктор не виртуальный, то вывод будет ~A()
